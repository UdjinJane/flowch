def decode_elfish(text: str) -> str:
    # 袩褉芯胁械褉泻邪 薪邪 褉褍褔薪芯泄 锌褉懈薪褍写懈褌械谢褜薪褘泄 蟹邪锌褍褋泻 芯褌 袣褝锌邪
    force_decode = False
    if text.strip().startswith("/写械泻芯写"):
        text = text.replace("/写械泻芯写", "", 1).strip()
        force_decode = True

    # 袘褉芯薪懈褉芯胁邪薪薪邪褟 小懈-屑邪褌褉懈褑邪 锌械褉械泻芯写懈褉芯胁泻懈 QWERTY <-> 袡笑校袣袝袧
    layout = str.maketrans(
        r"qwertyuiop[]asdfghjkl;'zxcvbnm,.QWERTYUIOP{}ASDFGHJKL:\"ZXCVBNM<>`~",
        r"泄褑褍泻械薪谐褕褖蟹褏褗褎褘胁邪锌褉芯谢写卸褝褟褔褋屑懈褌褜斜褞袡笑校袣袝袧袚楔些袟啸歇肖蝎袙袗袩袪袨袥袛袞协携效小袦袠孝鞋袘挟褢衼"
    )
    
    # 小褔懈褌邪械屑 谢邪褌懈薪褋泻懈械 懈 褉褍褋褋泻懈械 斜褍泻胁褘 写谢褟 褌芯褔薪芯泄 芯褑械薪泻懈 泻芯薪褌械泻褋褌邪
    low_text = text.lower()
    has_lat = any(c in "qwertyuiopasdfghjklzxcvbnm" for c in low_text)
    has_rus = any(c in "邪斜胁谐写械褢卸蟹懈泄泻谢屑薪芯锌褉褋褌褍褎褏褑褔褕褖褗褘褜褝褞褟" for c in low_text)
    
    # 校小袥袨袙袠携 袙袣袥挟效袝袧袠携:
    # 1. 小褉邪斜芯褌邪谢 褉褍褔薪芯泄 屑邪褉泻械褉 袣褝锌邪 (/写械泻芯写) OR
    # 2. 袙 褌械泻褋褌械 械褋褌褜 谢邪褌懈薪懈褑邪, 薪芯 袩袨袥袧袨小孝鞋挟 芯褌褋褍褌褋褌胁褍械褌 泻懈褉懈谢谢懈褑邪 (褔懈褋褌邪褟 芯锌械褔邪褌泻邪)
    if force_decode or (has_lat and not has_rus):
        return text.translate(layout)
        
    return text
