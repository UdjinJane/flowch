import re

def decode_flowch_stream(text: str) -> str:
    """
    Изолированный программный шлюз ввода шхуны Flowch.
    Перехватывает эльфийский Кэпа, игнорируя легитимные термины и синтаксис.
    """
    src_chars = r"qwertyuiop[]asdfghjkl;'zxcvbnm,./QWERTYUIOP{}ASDFGHJKL:\"ZXCVBNM<>?`~"
    dst_chars = r"йцукенгшщзхъфывапролджэячсмитьбю.ЙЦУКЕНГШЩЗХЪФЫВАПРОЛДЖЭЯЧСМИТЬБЮ,ёЁ"
    layout = str.maketrans(src_chars, dst_chars)
    
    # Защищенная зона: термины, которые нельзя трогать ни при каких условиях
    protected_terms = {
        "comfyui", "cuda", "vram", "loss", "gpu", "core", "test", 
        "run", "haha", "lol", "python", "def", "import", "bfloat16"
    }
    
    # Разбиваем строго по пробелам, чтобы сохранить исходное форматирование
    tokens = text.split(" ")
    decoded_tokens = []
    
    for token in tokens:
        if not token:
            decoded_tokens.append("")
            continue
            
        # Выделяем чистое слово для проверки на технический сленг
        clean_word = re.sub(r"[^\w]", "", token).lower()
        
        # Если слово в защищенном списке или содержит явные признаки кода
        if clean_word in protected_terms or any(c in token for c in ["(", ")", "[", "]", "{", "}", "=", ":", "_"]):
            decoded_tokens.append(token)
        else:
            # Маппим раскладку, полностью сохраняя префиксы, суффиксы и знаки
            decoded_tokens.append(token.translate(layout))
            
    return " ".join(decoded_tokens)

import openai  # или любой клиент, через который ты дергаешь LM Studio
from flowch_interceptor import decode_flowch_stream

def send_message_to_mistral(cap_raw_input: str):
    """
    Командный мостик отправки запросов в LM Studio.
    """
    # ШАГ 1: Перехватываем эльфийский на лету до отправки в модель
    clean_text = decode_flowch_stream(cap_raw_input)
    
    # ШАГ 2: Формируем пакет для LM Studio API
    # Мистраль увидит уже ИСПРАВЛЕННЫЙ и чистый текст!
    messages = [
        {"role": "system", "content": "Ты — Senior Python Developer... (твой жесткий сиспромпт)"},
        {"role": "user", "content": clean_text}
    ]
    
    # ШАГ 3: Запрос к реактору в соседней каюте
    response = openai.ChatCompletion.create(
        model="local-model",  # твой мистраль в LM Studio
        messages=messages,
        temperature=0.0  # держим строгость ответов
    )
    
    return response.choices[0].message.content
