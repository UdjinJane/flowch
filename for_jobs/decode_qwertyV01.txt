import re

def decode_flowch_stream(text: str) -> str:
    """
    袠蟹芯谢懈褉芯胁邪薪薪褘泄 锌褉芯谐褉邪屑屑薪褘泄 褕谢褞蟹 胁胁芯写邪 褕褏褍薪褘 Flowch.
    袩械褉械褏胁邪褌褘胁邪械褌 褝谢褜褎懈泄褋泻懈泄 袣褝锌邪, 懈谐薪芯褉懈褉褍褟 谢械谐懈褌懈屑薪褘械 褌械褉屑懈薪褘 懈 褋懈薪褌邪泻褋懈褋.
    """
    src_chars = r"qwertyuiop[]asdfghjkl;'zxcvbnm,./QWERTYUIOP{}ASDFGHJKL:\"ZXCVBNM<>?`~"
    dst_chars = r"泄褑褍泻械薪谐褕褖蟹褏褗褎褘胁邪锌褉芯谢写卸褝褟褔褋屑懈褌褜斜褞.袡笑校袣袝袧袚楔些袟啸歇肖蝎袙袗袩袪袨袥袛袞协携效小袦袠孝鞋袘挟,褢衼"
    layout = str.maketrans(src_chars, dst_chars)
    
    # 袟邪褖懈褖械薪薪邪褟 蟹芯薪邪: 褌械褉屑懈薪褘, 泻芯褌芯褉褘械 薪械谢褜蟹褟 褌褉芯谐邪褌褜 薪懈 锌褉懈 泻邪泻懈褏 褍褋谢芯胁懈褟褏
    protected_terms = {
        "comfyui", "cuda", "vram", "loss", "gpu", "core", "test", 
        "run", "haha", "lol", "python", "def", "import", "bfloat16"
    }
    
    # 袪邪蟹斜懈胁邪械屑 褋褌褉芯谐芯 锌芯 锌褉芯斜械谢邪屑, 褔褌芯斜褘 褋芯褏褉邪薪懈褌褜 懈褋褏芯写薪芯械 褎芯褉屑邪褌懈褉芯胁邪薪懈械
    tokens = text.split(" ")
    decoded_tokens = []
    
    for token in tokens:
        if not token:
            decoded_tokens.append("")
            continue
            
        # 袙褘写械谢褟械屑 褔懈褋褌芯械 褋谢芯胁芯 写谢褟 锌褉芯胁械褉泻懈 薪邪 褌械褏薪懈褔械褋泻懈泄 褋谢械薪谐
        clean_word = re.sub(r"[^\w]", "", token).lower()
        
        # 袝褋谢懈 褋谢芯胁芯 胁 蟹邪褖懈褖械薪薪芯屑 褋锌懈褋泻械 懈谢懈 褋芯写械褉卸懈褌 褟胁薪褘械 锌褉懈蟹薪邪泻懈 泻芯写邪
        if clean_word in protected_terms or any(c in token for c in ["(", ")", "[", "]", "{", "}", "=", ":", "_"]):
            decoded_tokens.append(token)
        else:
            # 袦邪锌锌懈屑 褉邪褋泻谢邪写泻褍, 锌芯谢薪芯褋褌褜褞 褋芯褏褉邪薪褟褟 锌褉械褎懈泻褋褘, 褋褍褎褎懈泻褋褘 懈 蟹薪邪泻懈
            decoded_tokens.append(token.translate(layout))
            
    return " ".join(decoded_tokens)
