import re

def decode_stream(text: str) -> dict:
    """
    袩芯褌芯泻芯胁褘泄 写械褕懈褎褉邪褌芯褉 褋 懈褋锌褉邪胁谢械薪薪芯泄 谢芯谐懈泻芯泄.
    袨斜褉邪斜邪褌褘胁邪械褌:
      - 袣械泄褋-懈薪褋械薪褋懈胁薪褘械 Python-泻谢褞褔械胁褘械 褋谢芯胁邪 懈 屑械褌芯写褘
      - 小锌械褑褋懈屑胁芯谢褘 胁 薪邪褔邪谢械/泻芯薪褑械 褌芯泻械薪芯胁
      - 袠褋泻谢褞褔械薪懈褟 写谢褟 泻芯褉芯褌泻懈褏 谢邪褌懈薪褋泻懈褏 褋谢芯胁
      - 袗锌芯褋褌褉芯褎褘 懈 蟹邪锌褟褌褘械 胁薪褍褌褉懈 褋谢芯胁
    """
    src_chars = r"qwertyuiop[]asdfghjkl;'zxcvbnm,./QWERTYUIOP{}ASDFGHJKL:\"ZXCVBNM<>?`~"
    dst_chars = r"泄褑褍泻械薪谐褕褖蟹褏褗褎褘胁邪锌褉芯谢写卸褝褟褔褋屑懈褌褜斜褞.袡笑校袣袝袧袚楔些袟啸歇肖蝎袙袗袩袪袨袥袛袞协携效小袦袠孝鞋袘挟,褢衼"
    layout = str.maketrans(src_chars, dst_chars)

    py_keywords = {
        "class", "def", "return", "if", "else", "elif", "for", "while", "try",
        "except", "import", "from", "as", "in", "is", "not", "and", "or",
        "with", "assert", "pass", "none", "true", "false", "lambda", "yield"
    }

    tokens = re.split(r"(\s+)", text)
    inner_tokens = []
    for token in tokens:
        if not token.strip():
            continue
        subtokens = re.findall(r"[\w']+|,|\W", token)
        inner_tokens.extend(subtokens)

    decoded_tokens = []
    for token in inner_tokens:
        clean_token = re.sub(r"[^\w',]", "", token)
        original_prefix = re.sub(r"[\w',]+$", "", token)
        original_suffix = re.sub(r"^[\w',]*", "", token)

        is_code = (
            any(casefold in py_keywords for casefold in [clean_token.casefold()])
            or "__" in clean_token
            or any(c in ["(", ")", "[", "]", "{", "}", "=", ":"] for c in token)
        )

        has_rus = bool(re.search(r"[邪-褟褢衼泄褑褍泻械薪谐褕褖蟹褏褗褎褘胁邪锌褉芯谢写卸褝褟褔褋屑懈褌褜斜褞]", token))
        has_lat = any(ord(c) in layout for c in token.lower())

        if "'" in clean_token or "," in clean_token:
            parts = []
            split_char = "'" if "'" in clean_token else ","
            parts = re.split(rf"(?<={split_char})", clean_token)
            decoded_parts = []
            for part in parts[1::2]:
                decoded_part = part.translate(layout) if has_lat else part
                decoded_parts.append(decoded_part)

            token_content = split_char.join(decoded_parts)
            if original_prefix + token_content + original_suffix != token:
                decoded_tokens.append(original_prefix + token_content + original_suffix)
            else:
                decoded_tokens.append(token)
        elif (has_lat and not has_rus and not is_code):
            decoded_part = clean_token.translate(layout)
            if len(decoded_part) > 50:
                decoded_tokens.append(token)
            else:
                decoded_tokens.append(original_prefix + decoded_part + original_suffix)
        else:
            decoded_tokens.append(token)

    final_text = "".join(decoded_tokens).replace('芦 ', '芦').replace(' 禄', '禄')
    return {
        "decoded": final_text,
        "contains_code": any(kw.casefold() in text.lower()
                            for kw in py_keywords) or "def " in text.lower()
    }
