import re

def decode_stream(text: str) -> dict:
    """
    袦芯薪芯谢懈褌薪褘泄 写械褕懈褎褉邪褌芯褉 袡笑校-褉邪褋泻谢邪写泻懈 褋 芯斜褉邪斜芯褌泻芯泄 Python-泻芯写邪.
    小芯褏褉邪薪褟械褌 谢械谐懈褌懈屑薪褘泄 泻芯写 懈 褌械褏薪懈褔械褋泻懈械 褌械褉屑懈薪褘 斜械蟹 懈蟹屑械薪械薪懈泄.
    """
    src_chars = r"qwertyuiop[]asdfghjkl;'zxcvbnm,./QWERTYUIOP{}ASDFGHJKL:\"ZXCVBNM<>?`~"
    dst_chars = r"泄褑褍泻械薪谐褕褖蟹褏褗褎褘胁邪锌褉芯谢写卸褝褟褔褋屑懈褌褜斜褞.袡笑校袣袝袧袚楔些袟啸歇肖蝎袙袗袩袪袨袥袛袞协携效小袦袠孝鞋袘挟,褢衼"
    layout = str.maketrans(src_chars, dst_chars)

    # 袩邪褌褌械褉薪褘 写谢褟 懈写械薪褌懈褎懈泻邪褑懈懈 Python-泻芯写邪 懈 谢械谐懈褌懈屑薪褘褏 褌械褉屑懈薪芯胁
    code_patterns = [
        r"\bdef\b", r"\bimport\b", r"(", ")", r"\[|\]",  # 小懈薪褌邪泻褋懈褋
        r"\bpython\b", r"\bcuda\b", r"\bai\b",           # 孝械褏薪懈褔械褋泻懈械 褌械褉屑懈薪褘
        r"haha", r"lol",                                   # 小谢械薪谐
    ]

    # 孝芯泻械薪懈蟹邪褑懈褟 褋 褋芯褏褉邪薪械薪懈械屑 褉械谐懈褋褌褉邪 懈 蟹薪邪泻芯胁 锌褉械锌懈薪邪薪懈褟
    tokens = re.findall(r"(\\w+|[^\\w\\s])", text)
    decoded_tokens = []

    for token in tokens:
        clean_token = re.sub(r"[^\\w]", "", token).lower()
        is_code = any(re.search(pattern, token) for pattern in code_patterns)

        # 袛械泻芯写懈褉褍械屑 褌芯谢褜泻芯 械褋谢懈 褝褌芯 薪械 泻芯写 懈 褋芯写械褉卸懈褌 谢邪褌懈薪褋泻懈械 褋懈屑胁芯谢褘
        if not is_code and any(ord(c) in layout for c in token.lower()):
            decoded_tokens.append(token.translate(layout))
        else:
            decoded_tokens.append(token)

    final_text = " ".join(decoded_tokens).strip()
    return {
        "decoded": final_text,
        "contains_code": any(pattern in text for pattern in code_patterns)
    }
