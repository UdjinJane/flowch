def parse_releases(file_path):
    """
    Функция для парсинга файла с исходными данными и разделения на:
    - Хотфиксы (-hf-)
    - Релизы (product-release)

    Обрабатывает возможные ошибки в исходных данных:
    * Отсутствие закрывающих/открывающих скобок
    * Пустые строки или строки с метаинформацией (например, "Операции")
    * Неправильный формат версии
    """
    
    # 1. Прочитать файл и разделить на строки
    try:
        with open(file_path, 'r', encoding='utf-8') as file:
            lines = file.readlines()
    except Exception as e:
        print(f"Ошибка при чтении файла: {e}")
        return None, None

    # 2. Очистка данных от лишних символов и пробелов
    cleaned_lines = []
    for line in lines:
        stripped_line = line.strip()
        if not stripped_line or stripped_line.startswith(('Операции', 'Изменить порядок версий')):
            continue  # Пропустить пустые строки и метаинформацию
        cleaned_lines.append(stripped_line)

    # 3. Разделение на хотфиксы и релиза
    hotfixes = []
    releases = []

    for line in cleaned_lines:
        parts = line.split()
        if len(parts) < 2:
            continue  # Пропустить строки без необходимой информации

        release_name = parts[0]
        version_type = None

        # Поиск версии в скобках
        for part in parts:
            if '(' in part and ')' in part:
                version_type = part.replace('(', '').replace(')', '')
                break  # Берем первую найденную версию

        if not version_type:
            continue  # Пропустить, если тип версии не определен

        # Определение типа версии
        if '-hf-' in version_type.lower() or '-HF-' in version_type.lower():
            hotfixes.append(f"{release_name} ({version_type})")
        elif 'product-release' in version_type.lower():
            releases.append(f"{release_name} ({version_type})")

    # 4. Возврат результатов
    return hotfixes, releases

# Пример использования:
if __name__ == "__main__":
    file_path = "Z:\\flowch\\for_jobs\\source.txt"
    hotfixes, releases = parse_releases(file_path)
    
    if hotfixes and releases:
        print("# Хотфиксы (-hf-)\n")
        for hf in hotfixes:
            print(hf)

        print("\n# Product Releases\n")
        for rel in releases:
            print(rel)
			
Пояснения к алгоритму:
Чтение файла: Файл читается построчно с учетом возможных ошибок (например, отсутствие файла).
Очистка данных:
Удаляются пустые строки.
Пропускаются строки с "Операции" или "Изменить порядок версий".
Разделение на хотфиксы и релиза:
Ищется часть строки, содержащая скобки ().
Если в скобках есть -hf- или -HF-, строка относится к хотфиксам.
Если в скобках есть product-release, строка относится к релизам.
Обработка ошибок:
Пропускаются строки без определенного типа версии.
Учитываются возможные ошибки в формате (например, отсутствие закрывающей скобки).
Проверка и сохранение алгоритма:
Алгоритм можно сохранить в файл с расширением .py для дальнейшего использования.
В случае изменений в исходных данных алгоритм будет корректно обрабатывать их, даже если есть ошибки (например, отсутствие закрывающих скобок).
